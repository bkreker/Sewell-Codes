' Returns the Month name of the cell value to the left of it
=IF(MONTH(A2)=1,"January",IF(MONTH(A2)=2,"February",IF(MONTH(A2)=3,"March",IF(MONTH(A2)=4,"April",IF(MONTH(A2)=5,"May",IF(MONTH(A2)=6,"June",IF(MONTH(A2)=7,"July",IF(MONTH(A2)=8,"August",IF(MONTH(A2)=9,"September",IF(MONTH(A2)=10,"October",IF(MONTH(A2)=11,"November",IF(MONTH(A2)=12,"December"))))))))))))

' Returns the day of the week, starting with 1 = Sunday
=IF(WEEKDAY(A2)=1,"Sunday",IF(WEEKDAY(A2)=2,"Monday",IF(WEEKDAY(A2)=3,"Tuesday",IF(WEEKDAY(A2)=4,"Wednesday",IF(WEEKDAY(A2)=5,"Thursday",IF(WEEKDAY(A2)=6,"Friday",IF(WEEKDAY(A2)=7,"Saturday")))))))

' Returns the Quarter of the year in column A2
=ROUNDUP(MONTH(A2)/3,0)
